# Testing of Simple Progarm Inheritance Chain

Tested with:

* snarkos v3.7.1
* amareleo-chain v2.4.1
* leo v2.6.0 (built from code)

```BASH
# Start a clean chain especially if you test this 
# multiple times to avoid name conflicts
amareleo-chain clean
amareleo-chain start --keep-state

leo build --path ./kaxxa_father_pgm/
leo build --path ./kaxxa_son_pgm/

# Use well known private key and view key pre-loaded with credits
export NODEPK=APrivateKey1zkp8CZNn3yeCseEtxuVPbDCwSyhGW6yZKUYKfgXmcpoGPWH
export NODEVK=AViewKey1mSnpFFC8Mj4fXbK5YiWgZ3mjiV8CxA79bYNa8ymUpTrw
export NODEADDR=aleo1rhgdu77hgyqd3xjj8ucu3jj9r2krwz6mnzyd80gncr5fxcwlh5rsvzp9px

# Deploy root dependency
snarkos  developer  deploy  kaxxa_father_pgm.aleo  \
    --private-key "${NODEPK}" \
    --path "./kaxxa_father_pgm/build/" \
    --query "http://localhost:3030" \
    --broadcast "http://localhost:3030/testnet/transaction/broadcast" \
    --priority-fee 100000 --network 1

# Verify: http://localhost:3030/testnet/program/kaxxa_father_pgm.aleo
# Restart amareleo-chain
^C
amareleo-chain start --keep-state

# Deploy dependent program
snarkos  developer  deploy  kaxxa_son_pgm.aleo  \
    --private-key "${NODEPK}" \
    --path "./kaxxa_son_pgm/build/" \
    --query "http://localhost:3030" \
    --broadcast "http://localhost:3030/testnet/transaction/broadcast" \
    --priority-fee 100000 --network 1

# Verify: http://localhost:3030/testnet/program/kaxxa_son_pgm.aleo
# Restart amareleo-chain
^C
amareleo-chain start --keep-state

# Execute dependent program
snarkos  developer  execute  kaxxa_son_pgm.aleo  son_call  \
    13u32   9u32 \
    --private-key "${NODEPK}"  \
    --query "http://localhost:3030"  \
    --broadcast "http://localhost:3030/testnet/transaction/broadcast" \
    --priority-fee 1000000 --network 1

# Lookup the record generated by the dependent program
snarkos developer scan -v "${NODEVK}"  --last 50  --endpoint "http://localhost:3030" --network 1
```
